export const deepGet = (
  obj: Record<string, unknown>,
  path: string[]
): unknown => {
  if (path.length === 0) return obj
  const [head, ...rest] = path
  const next = obj[head]
  if (next === undefined) return undefined
  return deepGet(next as any, rest)
}
