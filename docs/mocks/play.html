export const resolveDynamicRoute = (
  pathname: string,
  routes: DynamicRoute[],
  locale?: string
): ResolvedRoute | null => {
  if (!pathname || routes.length === 0) return null
  const normalized = pathname.replace(/\/+$/, "") || "/"
  const segments = normalized.split("/").filter(Boolean)

  for (const route of routes) {
    const match = matchRoute(segments, route.pattern)
    if (!match) continue
    const params = extractParams(match, route.paramNames)
    if (route.locale && locale && route.locale !== locale) {
      continue
    }
    return {
      route,
      params,
      pathname: normalized,
      locale: route.locale ?? locale ?? "default",
    }
  }
  return null
}
